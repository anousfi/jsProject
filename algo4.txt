Insertion sort 

VAR tab : ARRAY_OF Integer;
VAR i,j,tampon1,tampon2 : Integer;

BEGIN
  FOR i FROM 1 TO tab.length-1 STEP 1 DO 
      tampon1=tab[i]
      j:=i-1
         While (tab[j]<tab[j+1]) DO 
         tab[j]:=tab[j+1];
         j--;
         ENDWHILE
         tab[j+1]:=tampon1
      ENDFOR
  ENDFOR
END
